t = np.linspace(0,5,500)
k_values = [1,2,3]
T_okoli = 20

df = pd.DataFrame(index=t, columns=k_values)
df.iloc[0,:] = 90
for k in k_values:
    T = df[k].values
    for i in range(N-1):
        dt = t[i+1] - t[i]
        T[i+1] = T[i] - dt*(T[i]-T_okoli)*k
ax = df.plot()
