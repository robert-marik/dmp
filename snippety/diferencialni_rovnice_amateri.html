import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
from scipy.integrate import solve_ivp

pocatecni_podminka = [0.1]
meze = [0,10]

def rovnice(t, x, h=0):
    r = 1
    K = 1
    return r*x*(1-x/K) - h

t=np.linspace(*meze, 100)  # graf reseni
reseni = solve_ivp(
                   rovnice,
                   meze,
                   pocatecni_podminka,
                   t_eval=t, ## zde je možné zadat pole hodnot, kde se mají určit řešení
                   )
